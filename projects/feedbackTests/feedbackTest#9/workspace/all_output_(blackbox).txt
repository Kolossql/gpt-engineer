ShiftRegister
{
    // Clock signal to trigger the shift operation
    input wire clk

    // Active-low reset signal to reset the module to initial state
    input wire reset

    // Data bit to be shifted
    input wire data

    // Shift enable signal to control the shifting operation
    input wire shift_enable
}
{
    // Stored data output
    output wire [7:0] stored_data
}
